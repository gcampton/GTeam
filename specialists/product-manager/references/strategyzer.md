let isNullOrUndefined = function(value) {
    return value === undefined || value === null;
};

let getLastURI = function(url) {
	let path = "";
  	
    if (isNullOrUndefined(url)) { return;}
  
  	path = new URL(url).pathname.split('/');

    return path[path.length-1];
};

document.addEventListener('click', function(e) {
    let target = e.target;
    let tracker, dataset, aEl = "";
	
  	aEl = target.closest("[data-tracker]");
  	dataset = isNullOrUndefined(aEl) ? null : aEl.dataset
	
    if (isNullOrUndefined(dataset)) {
    	aEl = target.closest(".link-wrapper") || target.closest(".button");
      	if (isNullOrUndefined(aEl) && isNullOrUndefined(aEl.href)){
        	return;
        }
      	else {
        	tracker = getLastURI(aEl.href);
        }
    } else {
    	tracker = aEl.dataset.tracker;
    }
  	
  	if (!isNullOrUndefined(tracker)){
      tracker = tracker.trim();
      if (tracker.length > 0 && !tracker.includes(".pdf")){
      mixpanel.track("button_clicked", {"target": tracker, "object_type": aEl.nodeName, "css_class_name": aEl.className, "page_title": document.title, "source":"webflow"});
      }
    }
  	
});
