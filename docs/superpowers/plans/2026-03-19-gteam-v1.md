# GTeam v1 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build GTeam — a Claude Code skill system providing job-based AI orchestration across professional domains (law, accounting, SEO, marketing, content) with a two-layer Specialist + Jobs architecture and no API key requirement.

**Architecture:** Specialists define domain expertise as standalone SKILL.md files plus a `methodology.md` block. Jobs embed methodology blocks at build time via `gen-skill-docs.ts`. The main `/gteam` entry point routes user goals to the right job or specialist automatically.

**Tech Stack:** Bun (runtime + build), TypeScript, gstack browse binary (git submodule from `https://github.com/gcampton/gstack`), `claude -p` subprocess for Tier 3 eval (Ollama fallback).

```typescript
// scripts/gen-skill-docs.ts
import { readFile, writeFile } from 'node:fs/promises'
import { join, dirname, basename } from 'node:path'
import { glob } from 'node:fs/promises' // bun has glob built in

export const PREAMBLE_TEXT = `> GTeam update check: \`cd ~/dev/1_myprojects/gteam && git pull && bun run build\`
> Autonomy mode: execute fully automatically. Only pause for decisions with meaningful consequences to the user.`

export function resolvePreamble(): string {
  return PREAMBLE_TEXT
}

/** Convert token name back to specialist directory name.
 *  SEO_METHODOLOGY → seo
 *  CONTENT_CREATOR_METHODOLOGY → content-creator
 *  SOCIAL_MEDIA_METHODOLOGY → social-media
 */
function tokenToDir(token: string): string | null {
  if (!token.endsWith('_METHODOLOGY')) return null
  const base = token.slice(0, -'_METHODOLOGY'.length) // strip suffix
  return base.toLowerCase().replace(/_/g, '-')
}

export async function resolveToken(token: string, rootDir: string): Promise<string> {
  if (token === 'PREAMBLE') return resolvePreamble()

  const dir = tokenToDir(token)
  if (!dir) throw new Error(`Unknown token: ${token}`)

  const methodologyPath = join(rootDir, 'specialists', dir, 'methodology.md')
  try {
    return await readFile(methodologyPath, 'utf-8')
  } catch {
    throw new Error(`Cannot resolve {{${token}}}: file not found at ${methodologyPath}`)
  }
}

export async function processTemplate(tmplPath: string, rootDir: string): Promise<string> {
  let content = await readFile(tmplPath, 'utf-8')

  // Find all {{TOKEN}} occurrences
  const tokenRegex = /\{\{([A-Z_]+)\}\}/g
  const tokens = [...content.matchAll(tokenRegex)].map(m => m[1])

  // Resolve each unique token (validate all before substituting)
  const resolved = new Map<string, string>()
  for (const token of new Set(tokens)) {
    resolved.set(token, await resolveToken(token, rootDir))
  }

  // Substitute
  content = content.replace(tokenRegex, (_, token) => resolved.get(token)!)
  return content
}

// CLI entry point
if (import.meta.main) {
  const isDryRun = process.argv.includes('--dry-run')
  const rootDir = join(import.meta.dir, '..')

  // Discover all .tmpl files
  const patterns = [
    'SKILL.md.tmpl',
    'specialists/*/SKILL.md.tmpl',
    'jobs/*/SKILL.md.tmpl',
  ]

  let hadDiff = false

  for (const pattern of patterns) {
    for await (const tmplPath of glob(pattern, { cwd: rootDir })) {
      const fullTmplPath = join(rootDir, tmplPath)
      const outputPath = fullTmplPath.replace(/\.tmpl$/, '')

      const generated = await processTemplate(fullTmplPath, rootDir)

      if (isDryRun) {
        let existing = ''
        try { existing = await readFile(outputPath, 'utf-8') } catch {}
        if (existing !== generated) {
          console.error(`STALE: ${outputPath} — run bun run gen:skill-docs`)
          hadDiff = true
        }
      } else {
        await writeFile(outputPath, generated)
        console.log(`Generated: ${outputPath}`)
      }
    }
  }

  if (isDryRun && hadDiff) process.exit(1)
}
```

```typescript
// test/skill-validation.test.ts
import { expect, test, describe } from 'bun:test'
import { readFile } from 'node:fs/promises'
import { join } from 'node:path'
import { glob } from 'node:fs/promises'

// Import browse command sets from gstack submodule
// Path: browse/browse/src/commands.ts (gstack repo has browse/ subdir)
import {
  READ_COMMANDS,
  WRITE_COMMANDS,
  META_COMMANDS,
} from '../browse/browse/src/commands'

const ROOT = join(import.meta.dir, '..')
const KNOWN_TOKENS = new Set(['PREAMBLE'])
const KNOWN_TOKEN_SUFFIXES = ['_METHODOLOGY']

function isKnownToken(token: string): boolean {
  if (KNOWN_TOKENS.has(token)) return true
  return KNOWN_TOKEN_SUFFIXES.some(suffix => token.endsWith(suffix))
}

const ALL_BROWSE_COMMANDS = new Set([
  ...READ_COMMANDS,
  ...WRITE_COMMANDS,
  ...(META_COMMANDS ?? []),
])

async function discoverSkills(): Promise<string[]> {
  const paths: string[] = []
  for (const pattern of ['SKILL.md', 'specialists/*/SKILL.md', 'jobs/*/SKILL.md']) {
    for await (const p of glob(pattern, { cwd: ROOT })) {
      paths.push(join(ROOT, p))
    }
  }
  return paths
}

async function discoverTemplates(): Promise<string[]> {
  const paths: string[] = []
  for (const pattern of ['SKILL.md.tmpl', 'specialists/*/SKILL.md.tmpl', 'jobs/*/SKILL.md.tmpl']) {
    for await (const p of glob(pattern, { cwd: ROOT })) {
      paths.push(join(ROOT, p))
    }
  }
  return paths
}

describe('Tier 1: Skill frontmatter', () => {
  test('all SKILL.md files have required frontmatter fields', async () => {
    const skills = await discoverSkills()
    expect(skills.length).toBeGreaterThan(0)

    for (const skillPath of skills) {
      const content = await readFile(skillPath, 'utf-8')
      const frontmatterMatch = content.match(/^---\n([\s\S]*?)\n---/)
      expect(frontmatterMatch, `${skillPath}: missing frontmatter`).toBeTruthy()

      const fm = frontmatterMatch![1]
      expect(fm, `${skillPath}: missing 'name'`).toContain('name:')
      expect(fm, `${skillPath}: missing 'version'`).toContain('version:')
      expect(fm, `${skillPath}: missing 'description'`).toContain('description:')
      expect(fm, `${skillPath}: missing 'allowed-tools'`).toContain('allowed-tools:')
    }
  })
})

describe('Tier 1: Browse command validation', () => {
  test('all $B commands in SKILL.md files exist in browse registry', async () => {
    const skills = await discoverSkills()
    const errors: string[] = []

    for (const skillPath of skills) {
      const content = await readFile(skillPath, 'utf-8')
      // Match: $B command-name (anything after $B up to whitespace/end)
      const matches = content.matchAll(/\$B\s+(\S+)/g)
      for (const match of matches) {
        const cmd = match[1]
        if (!ALL_BROWSE_COMMANDS.has(cmd)) {
          errors.push(`${skillPath}: unknown $B command '${cmd}'`)
        }
      }
    }

    expect(errors, errors.join('\n')).toHaveLength(0)
  })
})

describe('Tier 1: Template token validation', () => {
  test('all {{TOKEN}} in .tmpl files are known tokens', async () => {
    const templates = await discoverTemplates()
    const errors: string[] = []

    for (const tmplPath of templates) {
      const content = await readFile(tmplPath, 'utf-8')
      const matches = content.matchAll(/\{\{([A-Z_]+)\}\}/g)
      for (const match of matches) {
        const token = match[1]
        if (!isKnownToken(token)) {
          errors.push(`${tmplPath}: unknown token {{${token}}}`)
        }
      }
    }

    expect(errors, errors.join('\n')).toHaveLength(0)
  })
})
```

```typescript
// scripts/skill-check.ts
import { readFile, stat } from 'node:fs/promises'
import { join } from 'node:path'
import { glob } from 'node:fs/promises'

const ROOT = join(import.meta.dir, '..')

async function fileExists(path: string): Promise<boolean> {
  try { await stat(path); return true } catch { return false }
}

async function hasFrontmatter(content: string): Promise<boolean> {
  return /^---\n[\s\S]*?\n---/.test(content)
}

async function checkSkill(skillPath: string) {
  const relPath = skillPath.replace(ROOT + '/', '')
  const tmplPath = skillPath + '.tmpl'

  const [hasMd, hasTmpl] = await Promise.all([
    fileExists(skillPath),
    fileExists(tmplPath),
  ])

  let status = '✓'
  const issues: string[] = []

  if (!hasTmpl) issues.push('missing .tmpl')
  if (!hasMd) issues.push('missing generated .md')

  if (hasMd) {
    const content = await readFile(skillPath, 'utf-8')
    if (!await hasFrontmatter(content)) issues.push('missing frontmatter')
  }

  if (issues.length) status = '✗'
  const issueStr = issues.length ? ` [${issues.join(', ')}]` : ''
  console.log(`  ${status} ${relPath}${issueStr}`)
}

async function main() {
  console.log('\nGTeam Skill Health Dashboard\n')

  const patterns = [
    { label: 'Entry Point', glob: 'SKILL.md' },
    { label: 'Specialists', glob: 'specialists/*/SKILL.md' },
    { label: 'Jobs', glob: 'jobs/*/SKILL.md' },
  ]

  for (const { label, glob: pattern } of patterns) {
    console.log(`${label}:`)
    let count = 0
    for await (const p of glob(pattern, { cwd: ROOT })) {
      await checkSkill(join(ROOT, p))
      count++
    }
    if (count === 0) console.log('  (none yet)')
    console.log()
  }
}

main()
```

```bash
#!/usr/bin/env bash
set -e

SCRIPT_DIR="$(cd "$(dirname "${BASH_SOURCE[0]}")" && pwd)"
GSTACK_BINARY="$HOME/.claude/skills/gstack/browse/dist/browse"
LOCAL_BINARY_DIR="$SCRIPT_DIR/browse/dist"

# Step 1: Init submodule if needed
if [ ! -f "$SCRIPT_DIR/browse/browse/src/commands.ts" ]; then
  echo "Initializing browse submodule..."
  cd "$SCRIPT_DIR"
  git submodule update --init
fi

# Step 2: Build or symlink browse binary
mkdir -p "$LOCAL_BINARY_DIR"

if [ -f "$GSTACK_BINARY" ] && [ -x "$GSTACK_BINARY" ]; then
  echo "Found existing gstack browse binary — symlinking (skipping build)"
  ln -sf "$GSTACK_BINARY" "$LOCAL_BINARY"
else
  echo "Building browse binary from submodule source..."
  cd "$SCRIPT_DIR/browse"
  bun install --frozen-lockfile 2>/dev/null || bun install
  bun run build 2>/dev/null || bun build --compile browse/src/cli.ts --outfile dist/browse
  cd "$SCRIPT_DIR"
fi

echo "✓ Browse binary ready: $LOCAL_BINARY"
echo ""

# Step 3: Generate SKILL.md files
echo "Generating SKILL.md files..."
bun run build
echo ""

# Step 4: Verify skills
echo "Skills installed:"
for pattern in "SKILL.md" "specialists/*/SKILL.md" "jobs/*/SKILL.md"; do
  for skill in $SCRIPT_DIR/$pattern; do
    [ -f "$skill" ] && echo "  ✓ ${skill#$SCRIPT_DIR/}"
  done
done

echo ""
echo "GTeam is ready. Skills installed at: $SCRIPT_DIR"
echo "Add to your project: ln -s $SCRIPT_DIR .claude/skills/gteam"
```

```typescript
async function detectJudgeBackend(): Promise<'claude' | 'ollama'> {
  if (process.env.GTEAM_JUDGE === 'ollama') return 'ollama'
  try {
    const result = await Bun.spawn(['which', 'claude'], { stdout: 'pipe' }).exited
    return result === 0 ? 'claude' : 'ollama'
  } catch {
    return 'ollama'
  }
}

async function runJudge(backend: 'claude' | 'ollama', prompt: string): Promise<string> {
  if (backend === 'claude') {
    const proc = Bun.spawn(['claude', '-p', prompt], { stdout: 'pipe', stderr: 'ignore' })
    await proc.exited
    return await new Response(proc.stdout).text()
  } else {
    // Ollama fallback
    const model = process.env.GTEAM_OLLAMA_MODEL ?? 'llama3'
    const res = await fetch('http://localhost:11434/api/generate', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ model, prompt, stream: false }),
    })
    const data = await res.json() as { response: string }
    return data.response
  }
}

function computeHash(content: string): string {
  return createHash('sha256').update(content).digest('hex')
}

      const key = skillPath.replace(ROOT + '/', '')

      if (!forceAll && hashes[key] === hash) {
        console.log(`  SKIP ${key} (unchanged)`)
        continue
      }

      const skillName = skillPath.split('/').at(-2)!
      const prompt = buildJudgePrompt(skillName, content)
      console.log(`  EVAL ${key}`)

      const response = await runJudge(backend, prompt)
      let scores: Record<string, number> & { pass: boolean }

      try {
        const jsonMatch = response.match(/\{[\s\S]*\}/)
        scores = JSON.parse(jsonMatch?.[0] ?? '{}')
      } catch {
        failures.push(`${key}: could not parse judge response`)
        continue
      }

      if (!scores.pass) {
        const low = Object.entries(scores)
          .filter(([k, v]) => k !== 'pass' && typeof v === 'number' && v < 70)
          .map(([k, v]) => `${k}=${v}`)
        failures.push(`${key}: failed on [${low.join(', ')}]`)
      } else {
        newHashes[key] = hash
      }
    }

    await saveHashes(newHashes)
    expect(failures, failures.join('\n')).toHaveLength(0)
  }, 120_000) // 2 minute timeout — LLM calls take time
})
```
